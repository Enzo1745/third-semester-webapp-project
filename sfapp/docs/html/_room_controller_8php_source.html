57 #[Route('/charge/salles/ajouter', name: 'app_room_management')]
58 public function manage(Request $request, EntityManagerInterface $entityManager): Response
59 {
60 $newRoom = new Room();
61 $formAddRoom = $this->createForm(AddRoomType::class, $newRoom);
62 $formAddRoom->handleRequest($request);
63
64 if ($formAddRoom->isSubmitted() && $formAddRoom->isValid()) {
65 // Check if a room with the same room number already exists
66 $existingRoom = $entityManager->getRepository(Room::class)
67 ->findOneBy(['roomName' => $newRoom->getRoomName()]);
68
69 if ($existingRoom) {
70 // Add an error flash message
71 $this->addFlash('danger', 'Cette salle existe déja');
72 } else {
73 // Save the new room to the database
74 $entityManager->persist($newRoom);
75 $entityManager->flush();
76 // Add a success flash message
77 $this->addFlash('success', 'Salle ajouté avec succès');
78 }
79 // Redirect back to the room management page
80 return $this->redirectToRoute('app_room_management');
81 }
82
83 // Render the room management template with the form
84 return $this->render('room/roomManagement.html.twig', [
85 'formAddRoom' => $formAddRoom->createView(),
86 ]);
87 }
